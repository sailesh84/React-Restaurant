  getScaleColour(value: number) {
    let [highMaxR, highMaxG, highMaxB] = [248, 105, 107];
    let [lowMaxR, lowMaxG, lowMaxB] = [99, 190, 123];
    const [midMaxR, midMaxG, midMaxB] = [255, 235, 132];
    let max = this.max;
    let min = this.min;

    if (Math.round(max - min) === 0) { return `${highMaxR}, ${highMaxG}, ${highMaxB}`; }

    const midPoint = (max + min) / 2;
    if (value > midPoint) {
      [lowMaxR, lowMaxG, lowMaxB] = [midMaxR, midMaxG, midMaxB];
      min = midPoint;
    } else {
      [highMaxR, highMaxG, highMaxB] = [midMaxR, midMaxG, midMaxB];
      max = midPoint;
    }

    const percentage = Math.abs((value - min) / (max - min));

    const redVal = Math.round(lowMaxR + ((highMaxR - lowMaxR) * percentage));
    const greenVal = Math.round(lowMaxG + ((highMaxG - lowMaxG) * percentage));
    const blueVal = Math.round(lowMaxB + ((highMaxB - lowMaxB) * percentage));

    return `${redVal}, ${greenVal}, ${blueVal}`;
  }

  getBooleanColour(value: any) {
    if (value === true) {
      return `${0}, ${176}, ${80}`;
    } else if (value === false) {
      return `${203}, ${44}, ${48}`;
    } else {
      return 'none';
    }
  }

  getZones(): any[] {
    // Fails counter

    const tabCountFails = [];
    let nbRows = this.analysis.cols.length - 1;
    let minus = 0;
    this.analysis.cols[0].subcols[0].data.forEach((value, i) => {
      let countFails = 0;

      for (let j = 1; j < this.analysis.cols.length; j++) {
        let acceptable = this.analysis.cols[j].subcols[this.analysis.cols[j].subcols.length - 1].data;
        if (acceptable.length === 0) {
          acceptable = new Array(this.analysis.cols[0].subcols[0].data.length).fill('');
          minus++;
        }
        if (acceptable[i] === false) {
          countFails++;
        }
      }
      tabCountFails.push({date: this.analysis.rows[i], count: countFails});
    });
    minus = minus / this.analysis.cols[0].subcols[0].data.length;
    nbRows -= minus;

    // Zone generator

    const zones = [];

    let endDate = null;
    let color = '';
    let fillColor = '';

    for (let index = 0; index < tabCountFails.length - 1; index++) {
      if (index <= 0) {
        if (tabCountFails[index].count <= 0) {
          color = 'rgba(99, 190, 123, 1)';
          fillColor = 'rgba(99, 190, 123, 0.2)';
        } else if (tabCountFails[index].count > 0 && tabCountFails[index].count < nbRows) {
          color = 'rgba(255, 193, 7, 1)';
          fillColor = 'rgba(255, 193, 7, 0.2)';
        } else {
          color = 'rgba(248, 105, 107, 1)';
          fillColor = 'rgba(248, 105, 107, 0.2)';
        }
      } else {
        if (tabCountFails[index].count <= 0) {
          if (tabCountFails[index + 1].count <= 0) { // Case V -> V
            // nothing
          } else if (tabCountFails[index + 1].count > 0 && tabCountFails[index + 1].count < nbRows) { // Case V -> A
            endDate = tabCountFails[index].date;
            zones.push({value: endDate, fillColor, color});
            color = 'rgba(255, 193, 7, 1)';
            fillColor = 'rgba(255, 193, 7, 0.2)';
          } else  {  // Case V -> F
            endDate = tabCountFails[index].date;
            zones.push({value: endDate, fillColor, color});
            color = 'rgba(248, 105, 107, 1)';
            fillColor = 'rgba(248, 105, 107, 0.2)';
          }
        } else if (tabCountFails[index].count > 0 && tabCountFails[index].count < nbRows) {
          if (tabCountFails[index + 1].count <= 0) { // Case A -> V
            endDate = tabCountFails[index + 1].date;
            zones.push({value: endDate, fillColor, color});
            color = 'rgba(99, 190, 123, 1)';
            fillColor = 'rgba(99, 190, 123, 0.2)';
          } else if (tabCountFails[index + 1].count > 0 && tabCountFails[index + 1].count < nbRows) { // Case A -> A
            // nothing
          } else  {  // Case A -> F
            endDate = tabCountFails[index].date;
            zones.push({value: endDate, fillColor, color});
            color = 'rgba(248, 105, 107, 1)';
            fillColor = 'rgba(248, 105, 107, 0.2)';
          }
        } else {
          if (tabCountFails[index + 1].count <= 0) { // Case F -> V
            endDate = tabCountFails[index + 1].date;
            zones.push({value: endDate, fillColor, color});
            color = 'rgba(99, 190, 123, 1)';
            fillColor = 'rgba(99, 190, 123, 0.2)';
          } else if (tabCountFails[index + 1].count > 0 && tabCountFails[index + 1].count < nbRows) { // Case F -> A
            endDate = tabCountFails[index + 1].date;
            zones.push({value: endDate, fillColor, color});
            color = 'rgba(255, 193, 7, 1)';
            fillColor = 'rgba(255, 193, 7, 0.2)';
          } else  {  // Case F -> F
            // nothing
          }
        }
      }
    }
    const lastElt = tabCountFails[tabCountFails.length - 1];
    if (lastElt.count <= 0) {
      color = 'rgba(99, 190, 123, 1)';
      fillColor = 'rgba(99, 190, 123, 0.2)';
    } else if (lastElt.count > 0 && lastElt.count < nbRows) {
      color = 'rgba(255, 193, 7, 1)';
      fillColor = 'rgba(255, 193, 7, 0.2)';
    } else {
      color = 'rgba(248, 105, 107, 1)';
      fillColor = 'rgba(248, 105, 107, 0.2)';
    }
    zones.push({fillColor, color});
    return zones;
  }

  getSeries(): any[] {
    const series = [{
      color: '',
      name: '',
      data: this.analysis.cols[0].subcols[0].data.slice().map((item, index) => [this.analysis.rows[index], item]),
      type: 'areaspline',
      yAxis: 0,
      marker: {
        enabled: true,
        radius: 4.5
      },
    }];

    const nbRows = this.analysis.cols.length - 1;

    const temp = [];

    for (let i = 1; i < this.analysis.cols.length; i++) {
      const tempRow = {name: this.analysis.cols[i].title, values: []};
      let acceptable = this.analysis.cols[i].subcols[this.analysis.cols[i].subcols.length - 1].data;
      if (acceptable.length === 0) {
        acceptable = new Array(this.analysis.cols[0].subcols[0].data.length).fill('');
      }
      acceptable.forEach((item, j) => {
        if (item === undefined || item === null || item === '') {
          tempRow.values.push([this.analysis.rows[j], 0]); // completed
        } else {
          if (item === true) {
            tempRow.values.push([this.analysis.rows[j], 1]); // passed
          } else {
            tempRow.values.push([this.analysis.rows[j], 2]); // failed
          }
        }
      });
      temp.push(tempRow);
    }

    temp.reverse().forEach((row) => {
      series.push({
        name: row.name,
        data: row.values.slice().map((item) => {
          let color;
          if (item[1] === 0) {
            color = '#34495e';
          } else if (item[1] === 1) {
            color = '#90ed7d';
          } else {
            color = '#f15c80';
          }
          return {x: item[0], y: - (0.7 / nbRows), color, borderWidth: 1, borderColor: 'white'};
        }),
        color: null,
        type: 'column',
        yAxis: 1,
        marker: null,
      });
    });

    return  series;
  }
