import {AfterViewInit, Component, ElementRef, OnDestroy, OnInit, ViewChild} from '@angular/core';
import {Notification} from '@app/shared/models/notification';
import {DataTableDirective} from 'angular-datatables';
import {Observable, Subject} from 'rxjs';
import {NgbActiveModal, NgbModal} from '@ng-bootstrap/ng-bootstrap';
import {NotificationsService} from '@app/core/services/notifications.service';
import {TitlePageService} from '@app/core/services/title-page.service';
import {ChangeEvent} from '@ckeditor/ckeditor5-angular';
import * as DecoupledEditor from '@ckeditor/ckeditor5-build-decoupled-document';
import {Log} from '@app/shared/models/log';
import {UserSharingService} from '@app/core/services/user-sharing.service';
import {LogsService} from '@app/core/services/logs.service';
import {takeUntil} from 'rxjs/operators';

@Component({
  selector: 'app-notifications',
  templateUrl: './notifications.component.html',
  styleUrls: ['./notifications.component.scss']
})
export class NotificationsComponent implements OnInit, AfterViewInit, OnDestroy {
  title = 'Notifications management';

  resetCache = false;

  errorModal = false;
  errorMessageModal = null;

  editor = DecoupledEditor;

  config = {
    toolbar: [
      'heading', '|', 'fontSize', 'fontFamily', '|', 'bold', 'italic', 'strikethrough', 'underline', 'highlight', '|', 'alignment', '|',
      'numberedList', 'bulletedList', '|', 'link', 'blockQuote', 'insertTable', 'tableColumn', 'tableRow', 'mergeTableCells', '|', 'undo',
      'redo'
    ]
  };

  dtOptions: any = {};
  notifications: Notification[];

  notification = {_id: '-1', name: '', message: '', type: 0, action: 0};

  log: Log = { _id: '-1', date: 0, userAgent: '', severity: 0, user: '', sourceIP: '', message: '' };

  state = { isLoaded: false, canConnect: null };

  @ViewChild('modalView', { static: true }) modalView: ElementRef;
  @ViewChild('modalAdd', { static: true }) modalAdd: ElementRef;
  @ViewChild('modalEdit', { static: true }) modalEdit: ElementRef;
  @ViewChild('modalDelete', { static: true }) modalDelete: ElementRef;

  @ViewChild(DataTableDirective, { static: true })
  dtElement: DataTableDirective;

  dtTrigger: Subject<any> = new Subject<any>();
  socketEvent: Observable<any>;

  private isSocketEventDead$ = new Subject();
  private isNotificationsDead$ = new Subject();
  private isLogsDead$ = new Subject();

  constructor(
    private modalService: NgbModal,
    private notificationsService: NotificationsService,
    private titlePageService: TitlePageService,
    private userSharingService: UserSharingService,
    private logsService: LogsService
  ) {
    this.titlePageService.setTitle(this.title);
  }

  ngOnInit() {
    const that = this;
    this.dtOptions = {
      responsive: true,
      pagingType: 'full_numbers',
      pageLength: 10,
      order: [[ 1, 'desc' ]],
      deferRender: true,
      ajax: (dataTablesParameters: any, callback) => {
        that.notificationsService.getNotifications(that.resetCache).pipe(takeUntil(this.isNotificationsDead$)).subscribe((response) => {
          that.notifications = response.data;
          that.state.canConnect = true;
          that.state.isLoaded = true;
          that.resetCache = false;
          callback({
            data: that.notifications
          });
        }, (error) => {
          that.state.canConnect = false;
          that.state.isLoaded = true;
          that.notifications = [];
          callback({
            data: that.notifications
          });
        });
      },
      columns: [{data: null}, { data: 'name' }, { data: 'message' }, { data: 'type' }, { data: 'action' }],
      columnDefs: [
        {
          orderable: false,
          targets: [0],
          className: 'td-action',
          render: (data, type, full, meta) => {
            return '<div class="btn-group btn-group-sm" role="group">' +
              '<button class="btn btn-outline-dark btn-view btn-sm"><i class="material-icons md-center md-sm">remove_red_eye</i>' +
              '</button>' +
              '<button class="btn btn-outline-info btn-edit btn-sm"><i class="material-icons md-center md-sm">edit</i></button>' +
              '<button class="btn btn-outline-danger btn-delete btn-sm"><i class="material-icons md-center md-sm">delete</i></button>' +
              '</div>';
          }
        },
        {
          targets: [1],
          render: (data, type, full, meta) => {
            return data.length > 20 ? data.substr( 0, 20 ) + '...' : data;
          }
        },
        {
          targets: [2],
          render: (data, type, full, meta) => {
            return data.length > 200
              ? data.substr(0, 200).replace(/\s([^\s]*)$/, '') + '...'
              : data;
          }
        },
        {
          targets: [3],
          render: (data, type, full, meta) => {
            return that.getBadges('type', data);
          }
        },
        {
          targets: [4],
          render: (data, type, full, meta) => {
            return that.getBadges('action', data);
          }
        },
      ],
      dom:
        '<"row"<"col-sm-3 col-12"l><"col-sm-6 col-8 text-center"B><"col-sm-3 col-12"f>>' +
        '<"row"<"col-sm-12 col-12"tr>>' +
        '<"row"<"col-sm-5 col-12"i><"col-sm-7 col-12"p>>',
      stateSave: true,
      buttons: [
        {
          extend: 'colvis',
          className: 'btn-outline-success btn-sm',
          init: (api, node, config) => {
            $(node).removeClass('btn-secondary');
          }
        }, {
          text: '<i class="material-icons md-center">add</i> Add',
          key: '1',
          className: 'btn-outline-success btn-sm',
          action: (e, dt, node, config) => {
            this.notification = {_id: '-1', name: '', message: '', type: 0, action: 0};
            this.modalService.open(this.modalAdd, { centered: true, size: 'xl', backdrop: 'static', scrollable: true });
          },
          init: (api, node, config) => {
            $(node).removeClass('btn-secondary');
          }
        }
      ],
      rowCallback: (row: Node, data: Notification, index: number) => {
        const self = this;
        // Unon first in order to avoid any duplicate handler
        // (see https://github.com/l-lin/angular-datatables/issues/87)
        const eltView = $('td', row).find('button.btn-view');
        if (eltView) {
          eltView.off('click');
          eltView.on('click', () => {
            that.notification = Object.assign({}, data);
            that.modalService.open(that.modalView, { centered: true, size: 'xl', backdrop: 'static', scrollable: true });
          });
        }
        const eltEdit = $('td', row).find('button.btn-edit');
        if (eltEdit) {
          eltEdit.off('click');
          eltEdit.on('click', () => {
            that.notification = Object.assign({}, data);
            that.modalService.open(that.modalEdit, { centered: true, size: 'xl', backdrop: 'static', scrollable: true });
          });
        }
        const eltDelete = $('td', row).find('button.btn-delete');
        if (eltDelete) {
          eltDelete.off('click');
          eltDelete.on('click', () => {
            that.notification = Object.assign({}, data);
            that.modalService.open(that.modalDelete, { centered: true, size: 'lg', backdrop: 'static' });
          });
        }
        return row;
      }
    };

    this.socketEvent = this.notificationsService.socketEvent;
  }

  ngAfterViewInit(): void {
    this.dtTrigger.next();
    this.socketEvent.pipe(takeUntil(this.isSocketEventDead$)).subscribe(() => {
      this.resetCache = true;
      this.refresh();
    });
  }

  ngOnDestroy(): void {
    // Do not forget to unsubscribe the event
    this.dtTrigger.unsubscribe();
    this.isSocketEventDead$.next();
    this.isNotificationsDead$.next();
    this.isLogsDead$.next();
  }

  onChangeCkeditor({ editor }: ChangeEvent ) {
    this.notification.message = editor.getData();
  }

  onReady( editor ) {
    editor.ui.getEditableElement().parentElement.insertBefore(
      editor.ui.view.toolbar.element,
      editor.ui.getEditableElement()
    );
  }

  refresh(): void {
    this.dtElement.dtInstance.then((dtInstance: DataTables.Api) => {
      // Destroy the table first
      dtInstance.destroy();
      // Call the dtTrigger to rerender again
      this.dtTrigger.next();
    });
  }

  verifyNotification() {
    for (const prop in this.notification) {
      if (prop !== '_id' && (this.notification[prop] === undefined || this.notification[prop] === null || this.notification[prop] === '')) {
        return false;
      }
    }
    return true;
  }


  addRow(modal: NgbActiveModal): void {
    // save in db
    if (this.verifyNotification() === true)  {
      this.notificationsService.addNotification(this.notification).pipe(takeUntil(this.isNotificationsDead$)).subscribe((response) => {
        this.notification = {_id: '-1', name: '', message: '', type: 0, action: 0};
        modal.close('Close click');
        const tempLog = Object.assign({}, this.log);
        const user = this.userSharingService.currentUser;
        tempLog.user = (user.firstname + ' ' + user.lastname).trim();
        tempLog.date = Date.now();
        tempLog.severity = 1;
        tempLog.message = `Added the notification ${response.data.name}`;
        this.logsService.addLog(tempLog).pipe(takeUntil(this.isLogsDead$)).subscribe();
      });
    } else {
      this.errorModal = true;
      this.errorMessageModal = 'Please complete all projects.';
    }
  }

  saveEditRow(modal: NgbActiveModal) {
    // save in db
    if (this.verifyNotification() === true)  {
      this.notificationsService.updateNotification(this.notification).pipe(takeUntil(this.isNotificationsDead$)).subscribe((response) => {
        this.notification = {_id: '-1', name: '', message: '', type: 0, action: 0};
        modal.close('Close click');
        const tempLog = Object.assign({}, this.log);
        const user = this.userSharingService.currentUser;
        tempLog.user = (user.firstname + ' ' + user.lastname).trim();
        tempLog.date = Date.now();
        tempLog.severity = 1;
        tempLog.message = `Updated the notification ${response.data.name}`;
        this.logsService.addLog(tempLog).pipe(takeUntil(this.isLogsDead$)).subscribe();
      });
    } else {
      this.errorModal = true;
      this.errorMessageModal = 'Please complete all projects.';
    }
  }

  deleteRow(modal: NgbActiveModal) {
    // save in db
    if (this.notification._id !== undefined || this.notification._id !== null || this.notification._id !== '') {
      this.notificationsService.deleteNotification(this.notification._id).pipe(takeUntil(this.isNotificationsDead$))
        .subscribe((response) => {
        const notifName = this.notification.name;
        this.notification = {_id: '-1', name: '', message: '', type: 0, action: 0};
        modal.close('Close click');
        const tempLog = Object.assign({}, this.log);
        const user = this.userSharingService.currentUser;
        tempLog.user = (user.firstname + ' ' + user.lastname).trim();
        tempLog.date = Date.now();
        tempLog.severity = 2;
        tempLog.message = `Deleted the notification ${notifName}`;
        this.logsService.addLog(tempLog).pipe(takeUntil(this.isLogsDead$)).subscribe();
      }, (error) => {
        this.errorModal = true;
        this.errorMessageModal = 'An error occurred on the server side.';
      });
    } else {
      this.errorModal = true;
      this.errorMessageModal = 'No notification to delete.';
    }
  }

  getBadges(type: string, value: number) {
    if (type === 'type') {
      if (value === 0) {
        return '<span class="badge badge-pill badge-primary">Email</span>';
      } else if (value === 1) {
        return '<span class="badge badge-pill badge-secondary">Push</span>';
      } else {
        return '<span class="badge badge-pill badge-success">SMS</span>';
      }
    } else {
      if (value === 0) {
        return '<span class="badge badge-pill badge-primary">Added / Created</span>';
      } else if (value === 1) {
        return '<span class="badge badge-pill badge-warning">Edited / Updated</span>';
      } else {
        return '<span class="badge badge-pill badge-danger">Deleted / Removed</span>';
      }
    }
  }

}
