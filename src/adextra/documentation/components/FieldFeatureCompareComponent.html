  getCalculateDifference(forecastOne, forecastTwo) {
    const waves = [];
    const swells = [];
    const windWaves = [];

    forecastOne.swell.forEach(item => {
      const isExistInForecastTwo = forecastTwo.swell.find(elt => elt[0] === item[0]);
      if (isExistInForecastTwo !== undefined) {
        swells.push([item[0], this.roundDecimal(item[1] - isExistInForecastTwo[1])]);
      }
    });
    forecastOne.wind_wave.forEach(item => {
      const isExistInForecastTwo = forecastTwo.wind_wave.find(elt => elt[0] === item[0]);
      if (isExistInForecastTwo !== undefined) {
        windWaves.push([item[0], this.roundDecimal(item[1] - isExistInForecastTwo[1])]);
      }
    });
    forecastOne.wave.forEach(item => {
      const isExistInForecastTwo = forecastTwo.wave.find(elt => elt[0] === item[0]);
      if (isExistInForecastTwo !== undefined) {
        waves.push([item[0], this.roundDecimal(item[1] - isExistInForecastTwo[1])]);
      }
    });

    return [waves, swells, windWaves];
  }

  roundDecimal(n: number, decimal: number = 2): number {
    const precision = Math.pow(10, decimal);
    return  Math.round( n * precision) / precision;
  }
