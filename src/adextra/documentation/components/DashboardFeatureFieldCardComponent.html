import {Component, Input, OnInit} from '@angular/core';
import {Project} from '@app/shared/models/field';
import {VesselType} from '@app/shared/models/vessel-type';

@Component({
  selector: 'app-dashboard-feature-field-card',
  templateUrl: './dashboard-feature-field-card.component.html',
  styleUrls: ['./dashboard-feature-field-card.component.scss']
})
export class DashboardFeatureProjectCardComponent implements OnInit {
  @Input() field: Field;
  @Input() vesselsTypes: VesselType[] = [];

  isCollapsed = true;
  expandIcon = 'expand_more';
  upp = Date.now();

  // types: { id: string, label: string, color: string }[] = [
  //   { id: '5d302a7be3c8ac10b44588d4', label: 'Reel-Lay', color: '#0a2f85' },
  //   { id: '5d302addda59a010b4334513', label: 'S-Lay', color: '#981e32' },
  //   { id: '5d302af1da59a010b4334514', label: 'Construction', color: '#7ab800' },
  //   { id: '5d302b15da59a010b4334515', label: 'Dive Support', color: '#3fcfd5' },
  //   { id: '5d302b29da59a010b4334516', label: 'Charter', color: '#e98300' }
  // ];

  constructor() { }

  ngOnInit() {
  }

  collapsePanel() {
    this.isCollapsed = !this.isCollapsed;
    if (this.isCollapsed) {
      this.expandIcon = 'expand_more';
    } else {
      this.expandIcon = 'expand_less';
    }
  }

  getBorderLeftColorVesselType(currentType: string) {
    const type = this.vesselsTypes.find((t) => t._id.toString() === currentType.toString());
    if (type !== undefined) {
      return type.color;
    }
  }
}
