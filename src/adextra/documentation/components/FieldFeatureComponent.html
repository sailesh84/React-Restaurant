import {Component, Input, OnInit} from '@angular/core';
import {ProjectsService} from '@app/core/services/fields.service';
import {Project} from '@app/shared/models/field';
import {User} from '@app/shared/models/user';
import {TitlePageService} from '@app/core/services/title-page.service';

@Component({
  selector: 'app-field-feature',
  templateUrl: './field-feature.component.html',
  styleUrls: ['./field-feature.component.scss']
})
export class ProjectFeatureComponent implements OnInit {
  @Input() fieldID = -1;
  @Input() vesselID = '-1';

  state = { isLoaded: false, canConnect: null };

  field: Project;
  title = ''; // Project Analysis
  forecastUpdated: Date;
  contact: User;

  constructor(private fieldsService: ProjectsService, private titlePageService: TitlePageService) {
    this.titlePageService.setTitle('Project Analysis');
  }

  ngOnInit() {
    this.getField(this.fieldID);
  }

  getField(id: number) {
    this.fieldsService.getField(id).subscribe(response => {
      this.field = response.data;

      this.state.canConnect = true;
      this.state.isLoaded = true;

      this.title += this.field.name;
      this.forecastUpdated = new Date();
      this.contact = this.field.vessels.find((elt) => elt._id === this.vesselID).contact;
    });
  }

}
