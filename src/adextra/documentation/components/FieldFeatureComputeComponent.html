import {AfterViewInit, Component, Input, OnInit} from '@angular/core';
import {AnalysisService} from '@app/core/services/analysis.service';
import {Analysis} from '@app/shared/models/analysis';
import * as Highcharts from 'highcharts';
import HC_boost from 'highcharts/modules/boost';
import HC_exporting from 'highcharts/modules/exporting';
import HC_offline_exporting from 'highcharts/modules/offline-exporting';
import HC_no_data_to_display from 'highcharts/modules/no-data-to-display';
import HC_accessibility from 'highcharts/modules/accessibility';
import HC_pattern_fill from 'highcharts/modules/pattern-fill';
import {Observable} from 'rxjs';
HC_boost(Highcharts);
HC_exporting(Highcharts);
HC_offline_exporting(Highcharts);
HC_no_data_to_display(Highcharts);
HC_accessibility(Highcharts);
HC_pattern_fill(Highcharts);

@Component({
  selector: 'app-field-feature-compute',
  templateUrl: './field-feature-compute.component.html',
  styleUrls: ['./field-feature-compute.component.scss']
})
export class FieldFeatureComputeComponent implements OnInit, AfterViewInit {
  @Input() fieldID = -1;
  @Input() vesselID = '-1';

  state = { isLoaded: false, canConnect: null };

  analysis: Analysis;
  highcharts = Highcharts;
  updateFlag = true;
  oneToOneFlag = true;
  // chartOptions = {};
  optionsGeneratedCharts: any[] = [];

  socketEvent: Observable<any>;

  constructor(private analysisService: AnalysisService) { }

  ngOnInit() {
    this.getAnalysis(this.fieldID, this.vesselID);
    this.socketEvent = this.analysisService.socketEvent;
  }

  ngAfterViewInit(): void {
    this.socketEvent.subscribe(() => {
      this.optionsGeneratedCharts = [];
      this.getAnalysis(this.fieldID, this.vesselID, true);
    });
  }

  getAnalysis(field, vessel, reset: boolean = false): void {
    this.analysisService.getAnalysis(field, vessel).subscribe(response => {
      this.analysis = response.data;
      this.state.canConnect = true;
      this.state.isLoaded = true;

      const that = this;

      for (let i = 1; i < this.analysis.cols.length; i++) {
        const options = {
          chart: {
            type: 'column',
            zoomType: 'x'
          },
          boost: {
            useGPUTranslations: true
          },
          noData: {
            style: {
              fontWeight: 'bold',
              fontSize: '15px',
              color: '#303030'
            }
          },
          title: {
            text: this.analysis.cols[i].title
          },
          xAxis: {
            crosshair: true,
            lineColor: '#000',
            lineWidth: 1,
            gridLineColor: '#000',
            pointInterval: 24 * 3600 * 1000,
            tickPixelInterval: 50,
            plotBands: this.getPlotBands(this.analysis),
            type: 'datetime',
            dateTimeLabelFormats: {
              day: '%a %e %b %H:%M'
            }
          },
          yAxis: {
            lineColor: '#000',
            lineWidth: 1,
            gridLineColor: '#000',
            allowDecimals: true,
            tickPixelInterval: 50,
            className: 'highcharts-color-0',
            title: {
              text: ''
            },
            max: 2,
            min: 0,
            labels: {
              format: '{value:.1f}'
            },
            plotLines: [{
              value: 1,
              color: '#FF0000',
              dashStyle: 'shortdash',
              width: 5,
              label: {
                text: 'Limit'
              }
            }]
          },
          tooltip: {
            shared: true,
            useHTML: true,
            formatter() {
              const points = this.points;
              let message = '<span style="font-weight: bold;">' + Highcharts.dateFormat('%a %e %b %H:%M', this.x) + '</span><table>';

              points.forEach((point) => {
                const col = (point.y === -1) ? 'N/A' : point.y;
                message += '<tr><td><span style="color: ' + point.color + '">\u25CF</span> ' + point.series.name + ' : </td><td style="' +
                  'text-align: right">' + ((col > 1) ? '<b style="color: rgb(222, 73, 41);">' + col + '</b>' : '<b>' + col + '</b>') +
                  '</td></tr>';
              });

              message += '</table>';

              return message;
            }
          },
          plotOptions: {
            column: {
              borderRadius: 5,
              pointWidth: 15,
              pointPadding: 0.1,
            }
          },
          credits: {
            enabled: false
          },
          series: [],
          responsive: {
            rules: [{
              condition: {
                maxWidth: 500
              },
              chartOptions: {
              }
            }]
          }
        };
        const showCharts = new Array(this.analysis.cols[i].subcols.length - 1);
        const addOptions = true;
        const series = [];
        for (let j = 0; j < this.analysis.cols[i].subcols.length - 1; j++) {
          if (this.analysis.cols[i].subcols[j].data.length === 0) {
            showCharts[j] = false;
          } else {
            showCharts[j] = true;
            series.push({
              name: this.analysis.cols[i].subcols[j].title,
              data: this.analysis.cols[i].subcols[j].data.slice().map((item, index) => {
                if (item === '-') {
                  return [this.analysis.rows[index], -1];
                } else {
                  return [this.analysis.rows[index], item];
                }
              })
            });
          }
        }
        if (showCharts.find((elt) => elt === true) !== undefined) {
          if (addOptions === true) {
            options.series = series;
            this.optionsGeneratedCharts.push(options);
          }
        }
      }
    });
  }

  getPlotBands(data) {
    const size = data.rows.length;
    if (size <= 1) {
      return [];
    }

    const plotBands = [
      {
        from: data.rows[0],
        to: data.rows[0] + (24 * 3600 * 1000),
        // color: 'rgba(0, 100, 0, 0.2)',
        color: 'url(#highcharts-default-pattern-0)',
        label: {
          text: 'Day 1',
          style: {
            // color: '#53565a',
            color: '#000000',
            fontWeight: 'bold',
            fontSize: '18px'
          },
          y: -10
        }
      },
      {
        from: data.rows[0] + (24 * 3600 * 1000),
        to: data.rows[0] + ((24 * 3600 * 1000) * 2),
        // color: 'rgba(255, 215, 0, 0.2)',
        color: 'url(#highcharts-default-pattern-3)',
        label: {
          text: 'Day 2',
          style: {
            // color: '#53565a',
            color: '#000000',
            fontWeight: 'bold',
            fontSize: '18px'
          },
          y: -10
        }
      },
      {
        from: data.rows[0] + ((24 * 3600 * 1000) * 2),
        to: data.rows[0] + ((24 * 3600 * 1000) * 3),
        // color: 'rgba(138, 43, 226, 0.2)',
        color: 'url(#highcharts-default-pattern-2)',
        label: {
          text: 'Day 3',
          style: {
            // color: '#53565a',
            color: '#000000',
            fontWeight: 'bold',
            fontSize: '18px'
          },
          y: -10
        }
      },
      {
        from: data.rows[0] + ((24 * 3600 * 1000) * 3),
        to: data.rows[data.rows.length - 1],
        // color: 'rgba(139, 69, 19, 0.2)',
        color: 'url(#highcharts-default-pattern-1)',
        label: {
          text: 'After day 3',
          style: {
            // color: '#53565a',
            color: '#000000',
            fontWeight: 'bold',
            fontSize: '18px'
          },
          y: -10
        }
      }
    ];

    const lastDate = data.rows[data.rows.length - 1];
    const firstDate = data.rows[0];
    const day = (24 * 3600 * 1000);

    if (lastDate <= (firstDate + day)) {
      return plotBands.slice(0, 1);
    } else if (lastDate <= (firstDate + (2 * day))) {
      return plotBands.slice(0, 2);
    } else if (lastDate <= (firstDate + (3 * day))) {
      return plotBands.slice(0, 3);
    } else {
      return plotBands.slice();
    }
  }

  calculateAmp(cols: any[], dates: number[]) {
    const tabMixed = [];
    cols[0].data.forEach((item, i) => {
      const mix = [];
      mix.push(1 - ((item === '-' ? 1 : item)));
      for (let j = 1; j < cols.length - 1; j++) {
        mix.push(1 - ((cols[j].data[i] === '-' ? 1 : cols[j].data[i])));
      }
      const precision = Math.pow(10, 3);
      let compute = Math.round( Math.min(... mix) * precision) / precision;
      if (compute > 1) {
        compute = 1;
      }
      if (compute > -1 && compute < 0) {
        compute = -1;
      }
      tabMixed.push([dates[i], compute]);
    });
    return tabMixed;
  }
}
